### Event Triggers

```typescript
// Auto-trigger on web source save
ipcMain.handle('web-source:save', async (_, source) => {
  const saved = await webSourceRepo.save(source);

  // Queue for extraction (non-blocking)
  extractionQueue.enqueue({
    sourceId: saved.source_id,
    locid: saved.locid,
    priority: 'normal',
    extractTypes: ['dates', 'entities', 'summary', 'title']
  });

  return saved;
});

// Manual trigger from UI
ipcMain.handle('extraction:processSource', async (_, sourceId) => {
  return extractionQueue.enqueue({
    sourceId,
    priority: 'high',
    extractTypes: ['dates', 'entities', 'summary', 'title']
  });
});

// Batch trigger for backfill
ipcMain.handle('extraction:processBatch', async (_, options) => {
  const sources = await webSourceRepo.findUnprocessed(options.limit);
  for (const source of sources) {
    extractionQueue.enqueue({
      sourceId: source.source_id,
      priority: 'batch'
    });
  }
  return { queued: sources.length };
});
```

## 2. DATABASE SCHEMA ADDITIONS

```sql
-- Smart titles and summaries on web_sources
ALTER TABLE web_sources ADD COLUMN smart_title TEXT;
ALTER TABLE web_sources ADD COLUMN smart_summary TEXT;
ALTER TABLE web_sources ADD COLUMN extraction_status TEXT DEFAULT 'pending';
-- Values: pending, processing, complete, failed, needs_review
ALTER TABLE web_sources ADD COLUMN extraction_confidence REAL;
ALTER TABLE web_sources ADD COLUMN extracted_at TEXT;

-- Unified entity store
CREATE TABLE entities (
  entity_id TEXT PRIMARY KEY,
  entity_type TEXT NOT NULL, -- 'person', 'organization', 'location', 'event'
  canonical_name TEXT NOT NULL,
  aliases TEXT, -- JSON array: ["Mike Muraco", "Michael Muraco"]
  description TEXT, -- "Real estate investor, purchased Lafayette Hills in 2006"
  first_seen_source TEXT REFERENCES web_sources(source_id),
  mention_count INTEGER DEFAULT 1,
  confidence REAL DEFAULT 0.5,
  verified INTEGER DEFAULT 0,
  created_at TEXT DEFAULT (datetime('now')),
  updated_at TEXT DEFAULT (datetime('now'))
);

CREATE INDEX idx_entities_type ON entities(entity_type);
CREATE INDEX idx_entities_name ON entities(canonical_name);

-- Link entities to locations with roles
CREATE TABLE location_entities (
  locid TEXT REFERENCES locs(locid),
  entity_id TEXT REFERENCES entities(entity_id),
  role TEXT NOT NULL, -- 'owner', 'developer', 'architect', 'investor', 'mentioned'
  start_date TEXT, -- When they became associated
  end_date TEXT, -- When association ended
  confidence REAL DEFAULT 0.5,
  source_id TEXT REFERENCES web_sources(source_id),
  created_at TEXT DEFAULT (datetime('now')),
  PRIMARY KEY (locid, entity_id, role)
);

-- Enhanced timeline events
ALTER TABLE location_timeline ADD COLUMN smart_title TEXT;
ALTER TABLE location_timeline ADD COLUMN tldr TEXT; -- 1-2 sentence summary
ALTER TABLE location_timeline ADD COLUMN extraction_source TEXT; -- 'ollama', 'spacy', 'manual'
ALTER TABLE location_timeline ADD COLUMN extraction_confidence REAL;
ALTER TABLE location_timeline ADD COLUMN needs_review INTEGER DEFAULT 0;

-- Extraction audit log
CREATE TABLE extraction_log (
  log_id TEXT PRIMARY KEY,
  source_id TEXT REFERENCES web_sources(source_id),
  locid TEXT,
  provider TEXT NOT NULL, -- 'ollama', 'spacy', 'fallback'
  model TEXT, -- 'qwen2.5:32b', 'en_core_web_lg'
  input_tokens INTEGER,
  output_tokens INTEGER,
  processing_time_ms INTEGER,
  dates_found INTEGER DEFAULT 0,
  entities_found INTEGER DEFAULT 0,
  confidence_avg REAL,
  quality_score REAL, -- 0-1 based on validation checks
  error_message TEXT,
  created_at TEXT DEFAULT (datetime('now'))
);
```

### 6.2 Output Validation

```typescript
interface QualityChecks {
  // Date validation
  dates: {
    yearRange: [1800, 2100],           // Reject impossible years
    requireContext: true,               // Must have surrounding text
    minConfidence: 0.5,                 // Filter low-confidence
    rejectPatterns: [
      /\d+ to \d+/,                     // "110 to 130" ranges
      /\$[\d,]+/,                       // Currency
      /\d+:\d+/,                        // Times without AM/PM
      /\d{1,3}(?:,\d{3})+/,            // Formatted numbers
    ]
  },

  // Entity validation
  entities: {
    minNameLength: 2,                   // Reject single chars
    maxNameLength: 100,                 // Reject garbage strings
    requireCapitalization: true,        // Names should be capitalized
    rejectPatterns: [
      /^[0-9]+$/,                       // Pure numbers
      /^https?:\/\//,                   // URLs
      /^\s*$/,                          // Whitespace
    ]
  },

  // Summary validation
  summary: {
    minLength: 50,                      // At least 50 chars
    maxLength: 500,                     // At most 500 chars
    requireSentenceEnd: true,           // Must end with punctuation
  },

  // Title validation
  title: {
    minLength: 10,                      // At least 10 chars
    maxLength: 80,                      // At most 80 chars
    rejectPatterns: [
      /^https?:\/\//,                   // URLs
      /\|/,                             // Pipe separators
      /^(null|undefined|error)/i,       // Error strings
    ]
  }
}
```

### 6.3 Confidence Scoring

```typescript
function calculateConfidence(extraction: RawExtraction): number {
  let score = 0.5; // Base confidence

  // Positive signals
  if (extraction.hasExplicitDate) score += 0.15;      // "built in 1923"
  if (extraction.hasKeywordContext) score += 0.15;    // "established", "closed"
  if (extraction.matchesPattern) score += 0.10;       // MM/DD/YYYY format
  if (extraction.multipleCorroboration) score += 0.10; // Multiple sources agree

  // Negative signals
  if (extraction.isApproximate) score -= 0.10;        // "circa", "around"
  if (extraction.shortContext) score -= 0.05;         // Little surrounding text
  if (extraction.ambiguousCategory) score -= 0.10;    // Can't determine type
  if (extraction.fromFooter) score -= 0.15;           // Footer/nav text

  return Math.max(0, Math.min(1, score));
}
```

### 6.6 Fallback Chain

```typescript
async function extractWithFallback(source: WebSource): Promise<ExtractionResult> {
  const providers = [
    { name: 'ollama', priority: 1 },
    { name: 'spacy', priority: 2 },
    { name: 'rules', priority: 3 },
  ];

  for (const provider of providers) {
    try {
      const result = await extract(source, provider.name);

      // Validate result
      const validation = validateExtraction(result);
      if (validation.isValid) {
        return {
          ...result,
          provider: provider.name,
          qualityScore: validation.score,
        };
      }

      // Log validation failure, try next provider
      log.warn(`${provider.name} extraction failed validation`, validation.errors);

    } catch (error) {
      log.error(`${provider.name} extraction error`, error);
    }
  }

  // All providers failed
  return {
    success: false,
    error: 'All extraction providers failed',
    needsManualReview: true,
  };
}
```

## 9. SUCCESS METRICS

| Metric | Target |
|--------|--------|
| Extraction accuracy | >90% correct dates |
| False positive rate | <5% garbage dates |
| Processing time | <30s per source |
| Auto-approve rate | >60% high-confidence |
| Entity dedup rate | <10% duplicates |
| Title improvement | 100% under 80 chars |
| Summary coverage | 100% of sources |
