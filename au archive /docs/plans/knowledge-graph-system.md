## 3. DATABASE SCHEMA: KNOWLEDGE GRAPH

```sql
-- Tags system
CREATE TABLE tags (
  tag_id TEXT PRIMARY KEY,
  category TEXT NOT NULL,  -- 'type', 'era', 'status', 'features', 'access'
  name TEXT NOT NULL,      -- 'golf-course', '1920s', etc.
  display_name TEXT,       -- 'Golf Course', '1920s'
  description TEXT,
  color TEXT,              -- For UI if needed
  sort_order INTEGER DEFAULT 0
);

-- Location tags (many-to-many)
CREATE TABLE location_tags (
  locid TEXT REFERENCES locs(locid),
  tag_id TEXT REFERENCES tags(tag_id),
  confidence REAL DEFAULT 0.5,
  source TEXT,  -- 'auto', 'manual', 'verified'
  created_at TEXT DEFAULT (datetime('now')),
  PRIMARY KEY (locid, tag_id)
);

CREATE INDEX idx_location_tags_tag ON location_tags(tag_id);

-- Entity connections (the spider web)
CREATE TABLE entity_connections (
  connection_id TEXT PRIMARY KEY,
  from_entity_id TEXT NOT NULL,
  from_entity_type TEXT NOT NULL,  -- 'location', 'person', 'organization'
  to_entity_id TEXT NOT NULL,
  to_entity_type TEXT NOT NULL,
  relationship TEXT NOT NULL,  -- 'designed_by', 'owned_by', 'same_era', 'nearby'
  strength REAL DEFAULT 0.5,   -- How strong is this connection
  start_date TEXT,
  end_date TEXT,
  source_id TEXT,  -- What source established this connection
  created_at TEXT DEFAULT (datetime('now'))
);

CREATE INDEX idx_connections_from ON entity_connections(from_entity_id, from_entity_type);
CREATE INDEX idx_connections_to ON entity_connections(to_entity_id, to_entity_type);
CREATE INDEX idx_connections_relationship ON entity_connections(relationship);

-- Pre-computed connection counts for fast querying
CREATE TABLE entity_connection_counts (
  entity_id TEXT NOT NULL,
  entity_type TEXT NOT NULL,
  connection_type TEXT NOT NULL,  -- 'person', 'organization', 'location'
  count INTEGER DEFAULT 0,
  PRIMARY KEY (entity_id, entity_type, connection_type)
);
```

## 4. KNOWLEDGE GRAPH QUERIES

### 4.1 "Show me all locations by this architect"

```sql
SELECT l.*
FROM locs l
JOIN location_entities le ON l.locid = le.locid
JOIN entities e ON le.entity_id = e.entity_id
WHERE e.canonical_name = 'Seymour Dunn'
  AND le.role = 'architect'
ORDER BY l.build_year;
```

### 4.2 "Show me all golf courses from the 1920s"

```sql
SELECT l.*
FROM locs l
JOIN location_tags lt1 ON l.locid = lt1.locid
JOIN location_tags lt2 ON l.locid = lt2.locid
JOIN tags t1 ON lt1.tag_id = t1.tag_id
JOIN tags t2 ON lt2.tag_id = t2.tag_id
WHERE t1.name = 'golf-course'
  AND t2.name = '1920s'
ORDER BY l.locnam;
```

### 4.3 "Show me locations connected to this one"

```sql
-- Direct connections (same owner, architect, etc.)
SELECT DISTINCT l.*, ec.relationship, e.canonical_name as connected_via
FROM locs l
JOIN entity_connections ec ON l.locid = ec.to_entity_id AND ec.to_entity_type = 'location'
JOIN entities e ON ec.from_entity_id = e.entity_id
WHERE ec.from_entity_id IN (
  SELECT le.entity_id
  FROM location_entities le
  WHERE le.locid = :this_locid
)
AND l.locid != :this_locid;
```

### 4.4 "Who owned the most abandoned places?"

```sql
SELECT
  e.canonical_name,
  e.entity_type,
  COUNT(DISTINCT le.locid) as location_count,
  GROUP_CONCAT(l.locnam, ', ') as locations
FROM entities e
JOIN location_entities le ON e.entity_id = le.entity_id
JOIN locs l ON le.locid = l.locid
WHERE le.role IN ('owner', 'developer', 'investor')
GROUP BY e.entity_id
ORDER BY location_count DESC
LIMIT 20;
```
